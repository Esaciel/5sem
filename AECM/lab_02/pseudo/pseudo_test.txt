int len = 8;                  // размер массива
int enroll = 4;               // обрабатываемых элементов за итерацию
int arr[len] = {1,2,3,4,5,6,7,8};
int sum = 0;

for (int i = 0; i < len; i += enroll) {
    sum += arr[i];
    sum += arr[i + 1];
    sum += arr[i + 2];
    sum += arr[i + 3];
}

sum += 1;                    // добавление 1 к итоговой сумме

while (1);                   // бесконечный цикл
